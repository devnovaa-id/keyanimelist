js/api.js

class AnimeAPI {
    constructor() {
        this.baseURL = 'https://api.ryzumi.vip/api/otakudesu';
        this.cache = new Map();
        this.cacheDuration = 5 * 60 * 1000; // 5 minutes cache
    }

    async request(endpoint, options = {}) {
        const cacheKey = endpoint + JSON.stringify(options);
        const cached = this.cache.get(cacheKey);
        
        if (cached && Date.now() - cached.timestamp < this.cacheDuration) {
            return cached.data;
        }

        try {
            const response = await fetch(`${this.baseURL}${endpoint}`, {
                ...options,
                headers: {
                    'accept': 'application/json',
                    ...options.headers
                }
            });

            if (!response.ok) {
                throw new Error(`API Error: ${response.status}`);
            }

            const data = await response.json();
            
            // Cache the response
            this.cache.set(cacheKey, {
                data,
                timestamp: Date.now()
            });

            return data;
        } catch (error) {
            console.error('API Request Failed:', error);
            throw error;
        }
    }

    // Get anime list with pagination and filtering
    async getAnimeList(type = 'complete', page = 1, genre = '', search = '') {
        let endpoint = `/anime?type=${type}&page=${page}`;
        
        if (genre) {
            endpoint += `&genre=${genre}`;
        }
        
        if (search) {
            endpoint = `/anime?type=${type}&search=${encodeURIComponent(search)}&page=${page}`;
        }
        
        return this.request(endpoint);
    }

    // Get all genres
    async getGenres() {
        return this.request('/genre');
    }

    // Get anime detail by slug
    async getAnimeDetail(slug) {
        return this.request(`/anime-info?slug=${slug}`);
    }

    // Get episode information
    async getEpisode(slug) {
        return this.request(`/anime/episode?slug=${slug}`);
    }

    // Get streaming iframe
    async getIframe(content, nonce) {
        return this.request(`/get-iframe?content=${content}&nonce=${nonce}`);
    }

    // Get nonce for iframe
    async getNonce() {
        return this.request('/nonce');
    }

    // Get anime schedule
    async getSchedule() {
        return this.request('/jadwal');
    }

    // Get batch download links
    async getBatchDownload(slug) {
        return this.request(`/download/batch?slug=${slug}`);
    }

    // Search anime
    async searchAnime(query, type = 'complete') {
        return this.request(`/anime?type=${type}&search=${encodeURIComponent(query)}`);
    }

    // Clear cache
    clearCache() {
        this.cache.clear();
    }

    // Get cached data
    getFromCache(key) {
        const cached = this.cache.get(key);
        if (cached && Date.now() - cached.timestamp < this.cacheDuration) {
            return cached.data;
        }
        return null;
    }
}

const api = new AnimeAPI();

class Router {
    constructor() {
        this.routes = {};
        this.currentPath = '';
        this.params = {};
        this.init();
    }

    init() {
        // Listen for hash changes
        window.addEventListener('hashchange', () => this.handleRoute());
        
        // Listen for page load
        window.addEventListener('load', () => this.handleRoute());
    }

    addRoute(path, handler) {
        this.routes[path] = handler;
    }

    async handleRoute() {
        const hash = window.location.hash.substring(1);
        const [path, ...queryParams] = hash.split('?');
        const query = new URLSearchParams(queryParams.join('?'));

        // Parse query parameters
        this.params = {};
        for (const [key, value] of query.entries()) {
            this.params[key] = value;
        }

        // Extract dynamic parameters
        const pathSegments = path.split('/');
        const routeSegments = Object.keys(this.routes).find(route => {
            const routePathSegments = route.split('/');
            if (routePathSegments.length !== pathSegments.length) return false;
            
            for (let i = 0; i < routePathSegments.length; i++) {
                if (routePathSegments[i].startsWith(':')) {
                    const paramName = routePathSegments[i].substring(1);
                    this.params[paramName] = pathSegments[i];
                } else if (routePathSegments[i] !== pathSegments[i]) {
                    return false;
                }
            }
            return true;
        });

        this.currentPath = path;

        // Show loading
        showLoading();

        try {
            if (routeSegments && this.routes[routeSegments]) {
                await this.routes[routeSegments](this.params);
            } else if (this.routes[path]) {
                await this.routes[path](this.params);
            } else {
                // Default to home if no route matches
                await this.routes['/'](this.params);
            }
        } catch (error) {
            console.error('Route handling error:', error);
            await this.showError('Terjadi kesalahan saat memuat halaman');
        } finally {
            hideLoading();
        }
    }

    async showError(message) {
        const content = document.getElementById('app-content');
        content.innerHTML = `
            <div class="error-page text-center py-5">
                <i class="fas fa-exclamation-triangle fa-3x text-warning mb-3"></i>
                <h2 class="mb-3">Oops!</h2>
                <p class="mb-4">${message}</p>
                <a href="#/" class="btn btn-primary">
                    <i class="fas fa-home"></i> Kembali ke Beranda
                </a>
            </div>
        `;
    }

    navigate(path) {
        window.location.hash = path;
    }

    getParam(key) {
        return this.params[key];
    }

    getCurrentPath() {
        return this.currentPath;
    }
}

class Utils {
    static debounce(func, wait) {
        let timeout;
        return function executedFunction(...args) {
            const later = () => {
                clearTimeout(timeout);
                func(...args);
            };
            clearTimeout(timeout);
            timeout = setTimeout(later, wait);
        };
    }

    static formatDate(dateString) {
        if (!dateString) return '-';
        const date = new Date(dateString);
        return date.toLocaleDateString('id-ID', {
            day: 'numeric',
            month: 'long',
            year: 'numeric'
        });
    }

    static truncateText(text, maxLength = 100) {
        if (text.length <= maxLength) return text;
        return text.substring(0, maxLength) + '...';
    }

    static parseAnimeInfo(infoText) {
        const info = {};
        const lines = infoText.split('\n');
        
        lines.forEach(line => {
            const parts = line.split(':');
            if (parts.length >= 2) {
                const key = parts[0].trim().toLowerCase().replace(/ /g, '_');
                const value = parts.slice(1).join(':').trim();
                info[key] = value;
            }
        });
        
        return info;
    }

    static getCurrentPage() {
        const params = new URLSearchParams(window.location.hash.split('?')[1]);
        return parseInt(params.get('page')) || 1;
    }

    static updatePageTitle(title) {
        document.title = `${title} - KeyAnimeList`;
    }

    static scrollToTop() {
        window.scrollTo({ top: 0, behavior: 'smooth' });
    }

    static copyToClipboard(text) {
        navigator.clipboard.writeText(text).then(() => {
            Components.showToast('Tautan berhasil disalin!', 'success');
        }).catch(err => {
            console.error('Copy failed:', err);
            Components.showToast('Gagal menyalin tautan', 'danger');
        });
    }

    static formatBytes(bytes, decimals = 2) {
        if (bytes === 0) return '0 Bytes';
        const k = 1024;
        const dm = decimals < 0 ? 0 : decimals;
        const sizes = ['Bytes', 'KB', 'MB', 'GB'];
        const i = Math.floor(Math.log(bytes) / Math.log(k));
        return parseFloat((bytes / Math.pow(k, i)).toFixed(dm)) + ' ' + sizes[i];
    }

    static isValidImage(url) {
        return new Promise((resolve) => {
            const img = new Image();
            img.onload = () => resolve(true);
            img.onerror = () => resolve(false);
            img.src = url;
        });
    }

    static lazyLoadImages() {
        const images = document.querySelectorAll('img[data-src]');
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    const img = entry.target;
                    img.src = img.dataset.src;
                    img.removeAttribute('data-src');
                    observer.unobserve(img);
                }
            });
        });
        
        images.forEach(img => observer.observe(img));
    }
}
